DECLARE @Sincronizado BIT;

-- Comprobar el estado de las réplicas en el grupo Always On
IF EXISTS (
    SELECT 1
    FROM sys.dm_hadr_database_replica_states drs
    INNER JOIN sys.availability_replicas ar 
        ON drs.replica_id = ar.replica_id
    INNER JOIN sys.availability_groups ag 
        ON ar.group_id = ag.group_id
    WHERE drs.synchronization_state_desc != 'SYNCHRONIZED'
      OR drs.synchronization_health_desc != 'HEALTHY'
)
    SET @Sincronizado = 0;  -- False: Hay problemas de sincronización
ELSE
    SET @Sincronizado = 1;  -- True: Todo está sincronizado correctamente

-- Mostrar el resultado
SELECT 
    CASE 
        WHEN @Sincronizado = 1 THEN 'True' 
        ELSE 'False' 
    END AS Sincronizado;

-- Crear o reemplazar el procedimiento almacenado
create PROCEDURE Sp_Verificar_Sincronizacion_AlwaysOn
AS
BEGIN
    DECLARE @Sincronizado BIT;
    DECLARE @Mensaje NVARCHAR(MAX);
    DECLARE @Asunto NVARCHAR(255);

    -- Comprobar el estado de las réplicas en Always On
    IF EXISTS (
        SELECT 1
        FROM sys.dm_hadr_database_replica_states drs
        INNER JOIN sys.availability_replicas ar 
            ON drs.replica_id = ar.replica_id
        INNER JOIN sys.availability_groups ag 
            ON ar.group_id = ag.group_id
        WHERE drs.synchronization_state_desc != 'SYNCHRONIZED'
          OR drs.synchronization_health_desc != 'HEALTHY'
    )
        SET @Sincronizado = 0;  -- False: Hay problemas de sincronización
    ELSE
        SET @Sincronizado = 1;  -- True: Todo está sincronizado correctamente

    -- Si no está sincronizado, enviar un correo electrónico
    IF @Sincronizado = 0
    BEGIN
        -- Configurar el mensaje de correo
        SET @Mensaje = 'Se detectaron problemas de sincronización en Always On. Por favor, revisar las réplicas para asegurar la continuidad de los servicios.';

        -- Configurar el asunto con el nombre del servidor
        SET @Asunto = 'Alerta: Problemas de Sincronización en Always On - ' + @@servername;

        -- Enviar correo mediante Database Mail
        EXEC msdb.dbo.sp_send_dbmail
            @profile_name = 'opesqladmin',  -- Reemplaza con tu perfil de correo de Database Mail
            @recipients = 'ingenieriadedatos@supervielle.com.ar',  -- Cambia al correo destinatario
            @copy_recipients = 'pablo.morixe@supervielle.com.ar',  -- Cambia al correo(s) en copia
            @subject = @Asunto,  -- Asunto con el nombre del servidor
            @body = @Mensaje;
    END

    -- Mostrar el resultado
    SELECT 
        CASE 
            WHEN @Sincronizado = 1 THEN 'True' 
            ELSE 'False' 
        END AS Sincronizado;
END;
GO
