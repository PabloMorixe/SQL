USE tempdb;

DECLARE @LogFileName NVARCHAR(255);
DECLARE @SQLCommand NVARCHAR(MAX);

-- Cursor para recorrer cada archivo de registro de la TempDB
DECLARE logFileCursor CURSOR FOR
SELECT name
FROM sys.master_files
WHERE database_id = DB_ID('tempdb')-- AND type_desc = 'LOG';

OPEN logFileCursor;

FETCH NEXT FROM logFileCursor INTO @LogFileName;

WHILE @@FETCH_STATUS = 0
BEGIN
    -- Generar y ejecutar el comando SHRINK para cada archivo de registro
    SET @SQLCommand = 'DBCC SHRINKFILE (' + QUOTENAME(@LogFileName, '''') + ', 1);';
    print(@SQLCommand); --O EXECUTE
	print ('DBCC FREEPROCCACHE')
	print ('DBCC DROPCLEANBUFFERS')
	print ('DBCC FREESYSTEMCACHE ('+'''ALL'''+')')
	print ('DBCC FREESESSIONCACHE')


    FETCH NEXT FROM logFileCursor INTO @LogFileName;
END

CLOSE logFileCursor;
DEALLOCATE logFileCursor;
