create PROCEDURE Sp_SET_COMPATIBILITY_LEVEL
    @ExcludedDatabases NVARCHAR(MAX) = NULL -- Lista separada por comas con nombres de BD a excluir
AS
BEGIN
    SET NOCOUNT ON;

    -- Determinar el nivel de compatibilidad adecuado según la versión del servidor
    DECLARE @SQLVersion NVARCHAR(128) = CAST(SERVERPROPERTY('ProductVersion') AS NVARCHAR(128));
    DECLARE @MajorVersion INT = CAST(LEFT(@SQLVersion, CHARINDEX('.', @SQLVersion) - 1) AS INT);
    DECLARE @CompatibilityLevel INT;

    -- Establecer el nivel de compatibilidad basado en la versión principal
    IF @MajorVersion >= 16 -- SQL Server 2022
        SET @CompatibilityLevel = 160;
    ELSE IF @MajorVersion = 15 -- SQL Server 2019
        SET @CompatibilityLevel = 150;
    ELSE IF @MajorVersion = 14 -- SQL Server 2017
        SET @CompatibilityLevel = 140;
    ELSE IF @MajorVersion = 13 -- SQL Server 2016
        SET @CompatibilityLevel = 130;
    ELSE IF @MajorVersion = 12 -- SQL Server 2014
        SET @CompatibilityLevel = 120;
    ELSE IF @MajorVersion = 11 -- SQL Server 2012
        SET @CompatibilityLevel = 110;
    ELSE IF @MajorVersion = 10 -- SQL Server 2008
        SET @CompatibilityLevel = 100;
    ELSE IF @MajorVersion = 9 -- SQL Server 2005
        SET @CompatibilityLevel = 90;
    ELSE 
        SET @CompatibilityLevel = 80; -- Nivel más bajo: SQL Server 2000

    PRINT 'El nivel de compatibilidad adecuado para esta versión del motor es: ' + CAST(@CompatibilityLevel AS NVARCHAR(3));

    -- Construir la lista de exclusiones si se proporcionó
    DECLARE @Exclusions TABLE (DatabaseName NVARCHAR(128));
    IF @ExcludedDatabases IS NOT NULL
    BEGIN
        INSERT INTO @Exclusions (DatabaseName)
        SELECT TRIM(value) 
        FROM STRING_SPLIT(@ExcludedDatabases, ',');
    END

    -- Cursor para recorrer solo las bases de datos con un nivel de compatibilidad diferente
    DECLARE db_cursor CURSOR FOR
    SELECT name 
    FROM sys.databases 
    WHERE state_desc = 'ONLINE' -- Solo bases de datos en línea
    AND compatibility_level <> @CompatibilityLevel -- Solo bases con compatibilidad diferente
    AND name NOT IN ('master', 'tempdb', 'model', 'msdb') -- Excluir bases del sistema
    AND (NOT EXISTS (SELECT 1 FROM @Exclusions WHERE DatabaseName = sys.databases.name) OR @ExcludedDatabases IS NULL); -- Excluir las especificadas

    -- Variables para almacenar el nombre de la base de datos
    DECLARE @DatabaseName NVARCHAR(128);

    -- Abrir el cursor
    OPEN db_cursor;
    FETCH NEXT FROM db_cursor INTO @DatabaseName;

    -- Ciclo para recorrer las bases de datos con compatibilidad diferente
    WHILE @@FETCH_STATUS = 0
    BEGIN
        -- Cambiar el nivel de compatibilidad
        DECLARE @SQL NVARCHAR(MAX);
        SET @SQL = 'ALTER DATABASE [' + @DatabaseName + '] SET COMPATIBILITY_LEVEL = ' + CAST(@CompatibilityLevel AS NVARCHAR(3)) + ';';

        EXEC sp_executesql @SQL;
        PRINT 'El nivel de compatibilidad de la base de datos [' + @DatabaseName + '] ha sido cambiado a ' + CAST(@CompatibilityLevel AS NVARCHAR(3)) + '.';

        -- Obtener la siguiente base de datos
        FETCH NEXT FROM db_cursor INTO @DatabaseName;
    END

    -- Cerrar y liberar el cursor
    CLOSE db_cursor;
    DEALLOCATE db_cursor;
END;

go
